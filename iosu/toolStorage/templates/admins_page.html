SELECT "storage_tools"."id", "storage_tools"."name", "storage_tools"."total_amount", "storage_tools"."wear_degree", COUNT("storage_production"."id") AS "usage_count" FROM "storage_tools" LEFT OUTER JOIN "storage_technologicalmapstools" ON ("storage_tools"."id" = "storage_technologicalmapstools"."tools_id") LEFT OUTER JOIN "storage_technologicalmap" ON ("storage_technologicalmapstools"."technological_map_id" = "storage_technologicalmap"."id") LEFT OUTER JOIN "storage_production" ON ("storage_technologicalmap"."id" = "storage_production"."technological_map_id") GROUP BY "storage_tools"."id" HAVING COUNT("storage_production"."id") > 0 ORDER BY "storage_tools"."wear_degree" ASC LIMIT 1

SELECT EXTRACT(MONTH FROM "storage_production"."start_date" AT TIME ZONE 'Europe/Moscow') AS "month", COUNT("storage_production"."id") AS "total_products" FROM "storage_production" INNER JOIN "storage_technologicalmap" ON ("storage_production"."technological_map_id" = "storage_technologicalmap"."id") INNER JOIN "storage_equipment" ON ("storage_technologicalmap"."equipment_id" = "storage_equipment"."id") WHERE "storage_equipment"."type_id" = 1 GROUP BY 1

equipment_union_tools = Equipment.objects.annotate( type_name=Value('Оборудование', output_field=CharField()) ).annotate( display_name=Cast('name', CharField()) ).values('type_name', 'display_name').union( Tools.objects.annotate( type_name=Value('Инструмент', output_field=CharField()) ).annotate( display_name=Concat('name', Value(' - '), 'wear_degree', Value(', '), 'total_amount', Value(' шт.'), output_field=CharField()) ).values('type_name', 'display_name'), all=True )
